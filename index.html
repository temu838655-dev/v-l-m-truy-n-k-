using System; using System.Collections.Generic; using UnityEngine; // Hệ thống Ngũ Hành chuẩn Võ Lâm public enum NguHanh { Kim, Moc, Thuy, Hoa_Tho } [System.Serializable] public class MonPhai { public string tenMonPhai; public NguHanh hePhai; public float cuongDoAtk; public string chieuThucDacTrung; } public class VoLamCharacter : MonoBehaviour { [Header("--- THÔNG TIN NHÂN VẬT ---")] public string tenHiepKhach = "Độc Cô Cầu Bại"; public MonPhai monPhaiHienTai; public int level = 1; public float hp = 500; public float maxHp = 500; public float satThuongCoBan = 50; [Header("--- ĐỐI TƯỢNG TƯƠNG TÁC ---")] public VoLamCharacter mucTieuChienDau; public bool laBoss = false; // Giả lập hệ thống Âm thanh & Hiệu ứng (Cần kéo AudioSource và ParticleSystem vào Unity) [Header("--- HIỆU ỨNG & ÂM THANH ---")] public AudioSource amThanhTungChieu; public ParticleSystem hieuUngVFX; void Start() { // Khởi tạo mặc định: Ví dụ chọn phái Cái Bang (Hệ Hỏa) monPhaiHienTai = new MonPhai { tenMonPhai = "Cái Bang", hePhai = NguHanh.Hoa_Tho, cuongDoAtk = 1.5f, chieuThucDacTrung = "Kháng Long Hữu Hối" }; Debug.Log($"[Võ Lâm] Khởi tạo nhân vật {tenHiepKhach} - Phái: {monPhaiHienTai.tenMonPhai}"); } // Hàm tung chiêu thức 3D chuẩn Võ Lâm public void TungChieuThuc() { if (mucTieuChienDau == null || mucTieuChienDau.hp <= 0) { Debug.Log("Không có mục tiêu hoặc mục tiêu đã tử trận!"); return; } // 1. Kích hoạt âm thanh sống động và hiệu ứng 3D if (amThanhTungChieu != null) amThanhTungChieu.Play(); if (hieuUngVFX != null) hieuUngVFX.Play(); // 2. Tính toán sát thương dựa trên Môn Phái và Ngũ Hành Khắc Chế float satThuongCuoi = satThuongCoBan * monPhaiHienTai.cuongDoAtk; // Giả lập tính năng khắc chế (Ví dụ: Hỏa khắc Kim) if (this.monPhaiHienTai.hePhai == NguHanh.Hoa_Tho && mucTieuChienDau.monPhaiHienTai?.hePhai == NguHanh.Kim) { satThuongCuoi *= 1.3f; // Tăng 30% sát thương do khắc hệ Debug.Log("[KHẮC CHẾ NGŨ HÀNH] Tăng 30% sát thương!"); } Debug.Log($"[{tenHiepKhach}] thi triển {monPhaiHienTai.chieuThucDacTrung} gây {satThuongCuoi} sát thương lên {mucTieuChienDau.tenHiepKhach}!"); // 3. Trừ máu mục tiêu mucTieuChienDau.NhanSatThuong(satThuongCuoi); } public void NhanSatThuong(float dmg) { hp -= dmg; if (hp <= 0) { hp = 0; if (laBoss) { Debug.Log($"🎉 BOSS {tenHiepKhach} đã bị tiêu diệt! Rơi ra Trang bị Hoàng Kim Môn Phái!"); } else { Debug.Log($"[{tenHiepKhach}] đã trọng thương, chuyển hướng về thành dưỡng sức."); } } } }
